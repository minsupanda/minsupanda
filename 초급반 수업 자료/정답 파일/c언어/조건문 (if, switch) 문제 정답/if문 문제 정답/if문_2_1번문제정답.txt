#include<stdio.h>
#include<Windows.h>

void main()
{
	int Kor, Eng, Math;
	printf("1.번문제\n");
	printf("세과목의 점수를 입력하시오\n");
	printf("국어 점수 : ");
	scanf("%d", &Kor);
	printf("수학 점수 : ");
	scanf("%d", &Math);
	printf("영어 점수 : ");
	scanf("%d", &Eng);

	int Sum = Kor + Math + Eng;
	float Avg = (float)Sum / 3.0f;

	char Grade;
	if (Avg >= 90)
		Grade = 'A';
	else if (Avg >= 80)
		Grade = 'B';
	else if (Avg >= 70)
		Grade = 'C';
	else if (Avg >= 60)
		Grade = 'D';
	else
		Grade = 'F';

	printf("\n국어점수 : %d\t수학점수 : %d\t\t영어점수 : %d\n총합점수 : %d\t평균점수 : %.2f\t등급 : %c\n", Kor, Math, Eng, Sum, Avg, Grade);
}
