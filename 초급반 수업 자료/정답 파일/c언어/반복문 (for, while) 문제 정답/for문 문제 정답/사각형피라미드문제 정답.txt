#include<stdio.h>

void main()
{
	int height, center = 0;
	printf("세로 입력 : ");
	scanf("%d", &height);
	center = height + 1;
	for (int y = 1; y <= height + 2; y++)
	{
		for (int x = 1; x <= height * 2 + 1; x++)
		{
			if (y == 1 || y == height + 2)
				printf("★");
			else if (x == 1 || x == height * 2 + 1)
				printf("★");
			else if (center - (y - 2) <= x && center + (y - 2) >= x)
				printf("★");
			else
				printf("  ");
		}
		printf("\n");
	}
}
