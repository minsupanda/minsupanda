#include <iostream>
#include <string>
#include <Windows.h>
#include<stdlib.h>
#include<time.h>
using namespace std;

void MergeSort(int Array[], int left, int mid, int right)
{
	int Left = left;
	int Right = mid + 1;

	int size = right - left + 1;
	int* temp;
	temp = new int[size];
	int t_index = 0;

	while (Left <= mid && Right <= right)
	{
		if (Array[Left] >= Array[Right])
			temp[t_index++] = Array[Right++];
		else
			temp[t_index++] = Array[Left++];
	}

	if (Left > mid)
	{
		while (Right <= right)
			temp[t_index++] = Array[Right++];
	}
	else
	{
		while (Left <= mid)
			temp[t_index++] = Array[Left++];
	}

	for (int i = 0; i < size; i++)
		Array[left + i] = temp[i];
	delete[] temp;
}

void Sort(int Array[], int left, int right)
{
	if (left >= right)
		return;
	int mid = (left + right) / 2;
	Sort(Array, left, mid);
	Sort(Array, mid + 1, right);
	MergeSort(Array, left, mid, right);
}

void Print_Num(int Array[], int size)
{
	for (int i = 0; i < size; i++)
		cout << i + 1 << "번 정수 값 : " << Array[i] << endl;
}

int* CreateNum(int Array[], int size)
{
	Array = new int[size];

	for (int i = 0; i < size; i++)
	{
		Array[i] = rand() % 100 + 1;
	}
	return Array;
}

void main()
{
	srand(time(NULL));
	int* Array = NULL;

	int size;
	cout << "할당할 정수 개수를 입력하세요 : ";
	cin >> size;
	Array = CreateNum(Array, size);

	cout << "정렬 전" << endl;
	Print_Num(Array, size);
	system("pause");

	Sort(Array, 0, size - 1);

	cout << "정렬 후" << endl;
	Print_Num(Array, size);
	system("pause");
	delete[] Array;
}
